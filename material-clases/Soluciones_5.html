6 calcular las correlaciones

múltiples y parciales

dec.corm            <- sqrt(1-1/diag(dec.inv))
pd                  <- length(dec.corm)  
cat("\n correlaciones múltiple de Decathlon \n")
## 
##  correlaciones múltiple de Decathlon
round(dec.corm,3)
##        X100m    Long.jump     Shot.put    High.jump        X400m X110m.hurdle       Discus   Pole.vault     Javeline       X1500m 
##        0.810        0.843        0.837        0.729        0.889        0.625        0.775        0.541        0.620        0.820
dec.part            <- dec.inv
for (i in 1:pd) {
  for (j in 1:(i-1)) {
   dec.part[i,j]   <- -dec.inv[i,j]/sqrt(dec.inv[i,i]*dec.inv[j,j])
  }
  dec.part[i,i]    <- dec.corm[i]
  dec.part[1:(i-1),i] <- dec.part[i,1:(i-1)]  
}
cat("\n correlaciones parciales de Decathlon \n")
